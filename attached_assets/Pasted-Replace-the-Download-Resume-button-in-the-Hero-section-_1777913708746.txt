⚙️ JavaScript Functions to Add
Add these two small functions to the existing <script> block — do not replace any existing JS:
javascriptfunction openResume() {
  const modal = document.getElementById('resume-modal');
  modal.style.display = 'flex';
  document.body.style.overflow = 'hidden';
}

function closeResume() {
  const modal = document.getElementById('resume-modal');
  modal.style.display = 'none';
  document.body.style.overflow = '';
}

// Close modal when clicking outside the iframe area
document.getElementById('resume-modal').addEventListener('click', function(e) {
  if (e.target === this) closeResume();
});

// Close modal with Escape key
document.addEventListener('keydown', function(e) {
  if (e.key === 'Escape') closeResume();
});
